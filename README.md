    data _null_;

     substring = "ABC";
     string = "cdeABCefgABCjklmnABC" ;

     /*----   Count down clock                                                ----*/
     position = -length(substring) + 1;

     if length(substring) <= length(string) then

         do until (position = 0);
         position = find(string,trim(substring), position+3);
         if position ne 0 then put position=;

     end;
     stop;
    run;
